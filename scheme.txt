CREATE TABLE spell (
    id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT, 
    name TEXT NOT NULL, 
    description TEXT NOT NULL, 
    short_description TEXT DEFAULT '', 
    type INTEGER NOT NULL DEFAULT 0,
    isRitual INTEGER NOT NULL DEFAULT 0, 
    isConcentration INTEGER NOT NULL DEFAULT 0, 
    hasVerbalComponent INTEGER NOT NULL DEFAULT 0, 
    hasSomaticComponent INTEGER NOT NULL DEFAULT 0, 
    hasMaterialComponent INTEGER NOT NULL DEFAULT 0, 
    materialComponent TEXT NOT NULL DEFAULT '', 
    range TEXT NOT NULL DEFAULT 'self', 
    duration TEXT NOT NULL DEFAULT 'Instantaneous', 
    time TEXT NOT NULL DEFAULT '1 action', 
    school TEXT NOT NULL DEFAULT 'Conjuration', 
    level INTEGER NOT NULL DEFAULT 0, 
    keywords TEXT NOT NULL DEFAULT ''
);
